Q. Assign a row number to the order we are fetching
SELECT 
    ROW_NUMBER() OVER(
        ORDER BY employees_id
    ) row_num, 
   employees_id, 
    first_name
FROM 
    employees;

SELECT 
    ROW_NUMBER() OVER(
        ORDER BY dept_code
    ) row_num, 
   count(employees_id),
 	dept_code
FROM 
    employees
group by dept_code


Q. Fetch only particular row / rows 


with eids_with_row as (
    SELECT 
    ROW_NUMBER() OVER(
        ORDER BY employees_id
    ) row_num, 
   employees_id, 
    first_name
FROM employees
)

SELECT * FROM eids_with_row
WHERE row_num >= 1 and row_num <= 10;
